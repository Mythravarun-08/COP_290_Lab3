62from . import db 

63from .models import User, Product, CartItems, OrderItems, SearchHistory 

64from lightfm import LightFM 

65from lightfm.data import Dataset 

66import numpy as np 

67import scipy.sparse as sp 

68 

69def recommendations(user_id): 

70 type_to_number = {'Television':0,'Mobile':1,'Laptop':2,'Tablet':3,'Gaming Console':4} 

71 

72 users = User.query.all() 

73 products = Product.query.all() 

74 

75 dataset = Dataset() 

76 dataset.fit(users=[u.id for u in users], items=[p.id for p in products]) 

77 

78 search_history = SearchHistory.query.all() 

79 search_final =[] 

80 for i in search_history: 

81 a = i.searches.split(',') 

82 a = list(map(int, a)) 

83 for k in a: 

84 if i.user_id < len(users) and k < len(products): 

85 search_final.append((i.user_id,k)) 

86 

87 if search_final: 

88 rows, cols = zip(*search_final) 

89 data = np.ones(len(rows)) 

90 

91 user_product_matrix = sp.coo_matrix((data, (rows, cols)), shape=(len(users), len(products))) 

92 for ci in CartItems.query.all(): 

93 user_product_matrix[ci.user_id, ci.product_id] = 1 

94 for oi in OrderItems.query.all(): 

95 user_product_matrix[oi.user_id, oi.product_id] = 1 

96 

97 product_features = np.zeros((len(products),7)) 

98 for i, product in enumerate(products): 

99 type = product.type 

100 if type in type_to_number: 

101 product_features[i,type_to_number[type]]=1 

102 product_features[i,5]=product.price 

103 # product_features[i,6]=product.rating 

104 

105 model = LightFM(loss='warp') 

106 model.fit(user_product_matrix, item_features=product_features, epochs=10) 

107 

108 user_index = dataset.mapping()[0][user_id] 

109 scores = model.predict(user_index,np.arange(len(products)),item_features=product_features) 

110 

111 top_products = sorted(zip(products,scores),key=lambda x:x[1],reverse=True)[:9] 

112 

113 return top_products
