57def gensalt(rounds: int = 12, prefix: bytes = b"2b") -> bytes: 

58 if prefix not in (b"2a", b"2b"): 

59 raise ValueError("Supported prefixes are b'2a' or b'2b'") 

60 

61 if rounds < 4 or rounds > 31: 

62 raise ValueError("Invalid rounds") 

63 

64 salt = os.urandom(16) 

65 output = _bcrypt.ffi.new("char[]", 30) 

66 _bcrypt.lib.encode_base64(output, salt, len(salt)) 

67 

68 return ( 

69 b"$" 

70 + prefix 

71 + b"$" 

72 + ("%2.2u" % rounds).encode("ascii") 

73 + b"$" 

74 + _bcrypt.ffi.string(output) 

75 )

78def hashpw(password: bytes, salt: bytes) -> bytes: 

79 if isinstance(password, six.text_type) or isinstance(salt, six.text_type): 

80 raise TypeError("Unicode-objects must be encoded before hashing") 

81 

82 if b"\x00" in password: 

83 raise ValueError("password may not contain NUL bytes") 

84 

85 # bcrypt originally suffered from a wraparound bug: 

86 # http://www.openwall.com/lists/oss-security/2012/01/02/4 

87 # This bug was corrected in the OpenBSD source by truncating inputs to 72 

88 # bytes on the updated prefix $2b$, but leaving $2a$ unchanged for 

89 # compatibility. However, pyca/bcrypt 2.0.0 *did* correctly truncate inputs 

90 # on $2a$, so we do it here to preserve compatibility with 2.0.0 

91 password = password[:72] 

92 

93 # When the original 8bit bug was found the original library we supported 

94 # added a new prefix, $2y$, that fixes it. This prefix is exactly the same 

95 # as the $2b$ prefix added by OpenBSD other than the name. Since the 

96 # OpenBSD library does not support the $2y$ prefix, if the salt given to us 

97 # is for the $2y$ prefix, we'll just mugne it so that it's a $2b$ prior to 

98 # passing it into the C library. 

99 original_salt, salt = salt, _normalize_re.sub(b"$2b$", salt) 

100 

101 hashed = _bcrypt.ffi.new("char[]", 128) 

102 retval = _bcrypt.lib.bcrypt_hashpass(password, salt, hashed, len(hashed)) 

103 

104 if retval != 0: 

105 raise ValueError("Invalid salt") 

106 

107 # Now that we've gotten our hashed password, we want to ensure that the 

108 # prefix we return is the one that was passed in, so we'll use the prefix 

109 # from the original salt and concatenate that with the return value (minus 

110 # the return value's prefix). This will ensure that if someone passed in a 

111 # salt with a $2y$ prefix, that they get back a hash with a $2y$ prefix 

112 # even though we munged it to $2b$. 

113 return original_salt[:4] + _bcrypt.ffi.string(hashed)[4:]

116def checkpw(password: bytes, hashed_password: bytes) -> bool: 

117 if isinstance(password, six.text_type) or isinstance( 

118 hashed_password, six.text_type 

119 ): 

120 raise TypeError("Unicode-objects must be encoded before checking") 

121 

122 if b"\x00" in password or b"\x00" in hashed_password: 

123 raise ValueError( 

124 "password and hashed_password may not contain NUL bytes" 

125 ) 

126 

127 ret = hashpw(password, hashed_password) 

128 

129 if len(ret) != len(hashed_password): 

130 return False 

131 

132 return _bcrypt.lib.timingsafe_bcmp(ret, hashed_password, len(ret)) == 0

135def kdf( 

136 password: bytes, 

137 salt: bytes, 

138 desired_key_bytes: int, 

139 rounds: int, 

140 ignore_few_rounds: bool = False, 

141) -> bytes: 

142 if isinstance(password, six.text_type) or isinstance(salt, six.text_type): 

143 raise TypeError("Unicode-objects must be encoded before hashing") 

144 

145 if len(password) == 0 or len(salt) == 0: 

146 raise ValueError("password and salt must not be empty") 

147 

148 if desired_key_bytes <= 0 or desired_key_bytes > 512: 

149 raise ValueError("desired_key_bytes must be 1-512") 

150 

151 if rounds < 1: 

152 raise ValueError("rounds must be 1 or more") 

153 

154 if rounds < 50 and not ignore_few_rounds: 

155 # They probably think bcrypt.kdf()'s rounds parameter is logarithmic, 

156 # expecting this value to be slow enough (it probably would be if this 

157 # were bcrypt). Emit a warning. 

158 warnings.warn( 

159 ( 

160 "Warning: bcrypt.kdf() called with only {0} round(s). " 

161 "This few is not secure: the parameter is linear, like PBKDF2." 

162 ).format(rounds), 

163 UserWarning, 

164 stacklevel=2, 

165 ) 

166 

167 key = _bcrypt.ffi.new("uint8_t[]", desired_key_bytes) 

168 res = _bcrypt.lib.bcrypt_pbkdf( 

169 password, len(password), salt, len(salt), key, len(key), rounds 

170 ) 

171 _bcrypt_assert(res == 0) 

172 

173 return _bcrypt.ffi.buffer(key, desired_key_bytes)[:]
